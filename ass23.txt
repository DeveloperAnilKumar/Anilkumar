10. Write a C++ program to add all the numbers of an array of size 10

#include <iostream>
using namespace std;
int main()
{
    int arr[] = {10, 20, 30, 40, 50, 15, 18, 19, 52};
    int sum = 0;

    for (int i = 0; i <= 10; i++)
    {
        sum = sum + arr[i];
    }
    cout << sum << endl;
    return 0;
}
